git branch <haaran nimi>
	-Luo uuden haaran HEADin osoittamaan committiin.
git checkout -b <haaran nimi>
	-Luo uuden haaran HEADin osoittamaan committiin ja siirtää HEADin osoittamaan
	 siihen.

git branch
	-Listaa paikalliset haarat. Aktiivinen haara on merkitty tähdellä (*).
git branch -a
	-Listaa kaikki paikalliset haarat ja etätietovarastojen haarat. Aktiivinen haara 
	 on merkitty tähdellä (*).

git branch -d <haaran nimi>
	-Poistaa nimetyn haaran paikallisesta tietovarastosta.

git checkout <haaran nimi>
	-Siirtää HEADin nimettyyn haaraan
git checkout <commit>
	-Siirtää HEADin committiin, jolloin ollaan detached-HEAD tilassa (eli HEAD on irti haarasta).

git diff HEAD <commit/haara>
	-Vertaa HEADia määriteltyyn committiin tai haaraan.
git diff <commit/haara 1> <commit/haara 2>
	-Vertaa kahta määriteltyä committia tai haaraa keskenään.
git diff HEAD <commit/haara> -- <polku>
	-Vertaa HEADia määriteltyyn committiin tai haaraan ainoastaan määritellystä polusta.

git reset <commit>
	-Siirtää HEADin osoittaman haaran osoittamaan nykyisen commitin sijaan määriteltyä committia.
git reset HEAD~<luku>
	-Siirtää HEADin luvun osoittaman määrän taaksepäin versiohistoriassa.
git reset --hard <commit>
	-Siirtää HEADin osoittaman haaran osoittamaan nykyisen commitin sijaan määriteltyä committia. 
	 Muuttaa työpuun vastaamaan committia.

git checkout -b <haaran nimi> <remote haaran nimi>
	-Luo uuden haaran määritellyn remote-haaran kohdalle. Siirtää HEADin luotuun haaraan. 
	 (Paikallisen haaran nimen ei tarvitse olla sama kuin remote-haaran nimi)
git branch <haaran nimi> <remote haaran nimi>
	-Luo uuden haaran määritellyn remote-haaran kohdalle. (Paikallisen haaran nimen ei tarvitse 
	 olla sama kuin remote-haaran nimi)
git checkout <remote haaran nimi>
	-Luo uuden haaran määritellyn remote-haaran nimen pohjalta. Siirtää HEADin luotuun haaraan. 
	 (Jos kyseessä on uniikki nimi ja riittävän uusi versio Gitistä)
git checkout --track <remote haaran nimi>
	-Luo uuden haaran määritellyn remote-haaran nimen pohjalta. Siirtää HEADin luotuun haaraan. 
	 Asettaa luodun haaran seuraamaan etätietovaraston haaraa.
